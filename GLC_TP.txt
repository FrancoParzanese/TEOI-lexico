LISTA_VAR= ID | ID COMA {LISTA_VAR}
LISTA_TIPO= {TIPO} | {LISTA_TIPO} COMA {TIPO}
TIPO= REAL | INTEGER| STRING
PROGRAMA= COMIENZO.PROGRAMA {BLOQUE_SENTENCIAS} FIN.PROGRAMA
BLOQUE_SENTENCIAS= {SENTENCIA} | {BLOQUE_SENTENCIA} {SENTENCIA}
SENTENCIA= {ASIGNACION} | {MIENTRAS}| {DECISIONES}| {SALIDA} | {COMENTARIOS}
ASIGNACION= ID IGUAL ({VAR_NUMERIC}|{VAR_STRING})
EXPRESION = {EXPRESION} (OP_SUMA|OP_RES) {TERMINO} | {TERMINO}
TERMINO = {TERMINO} (OP_MUL | OP_DIV) {VAR_NUMERIC} | {VAR_NUMERIC}
VAR_NUMERIC= {CONST_NUMERIC} | {EXPRESION}
VAR_STRING= CONST_STRING
CONST_NUMERIC= {CONST_INT} | {CONST_FLOAT}
CONST_FLOAT= DIGITO PUNTO DIGITO | PUNTO DIGITO
MIENTRAS= MIENTRAS {EXP_LOGICA} LLAV_ABRE {SENTENCIA} LLAV_CIE
EXP_LOGICA= {CONDICION} | {CONDICION} (AND|OR) {CONDICION}
CONDICION= {EXPRESION} (OP_MAY | OP_MEN| IGUAL| OP_MENIG| OP_MAYIG| OP_DIS) {EXPRESION}
DECISIONES= IF {EXP_LOGICA} THEN LLAV_ABRE {SENTENCIA} LLAV_CIE |
	    IF {EXP_LOGICA} THEN LLAV_ABRE {SENTENCIA} LLAV_CIE 
	    ELSE LLAV_ABRE {SENTENCIA} LLAV_CIE
SALIDA= PRINT {SENT_SALIDA}
SENT_SALIDA= CONST_STRING | CONST_NUMERIC
